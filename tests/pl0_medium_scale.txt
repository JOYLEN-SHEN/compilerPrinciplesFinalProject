const MAX = 50, PI = 3, E = 2;
var sum, count, temp, flag;
var result1, result2, result3;

begin
  sum := 0;
  count := 0;
  flag := 1;

  while count < MAX do
    begin
      temp := count * 2;
      if temp > 30 then
        begin
          if flag = 1 then
            sum := sum + temp
          else
            sum := sum - temp;
          flag := 1 - flag
        end
      else
        sum := sum + count;
      count := count + 1
    end;

  result1 := sum + PI;
  result2 := sum - E;
  result3 := (result1 + result2) * PI / E;

  if result3 > 100 then
    temp := result3 + 10
  else
    temp := result3 * 2
end.
