const MAX = 100, MIN = 0, STEP = 5;
var sum, avg, count, flag;
var result1, result2, result3;

function calculate(x: int, y: int): int {
  const local_const = 10;
  var local_var;
  local_var := x + y;
  return local_var * local_const;
}

function process(): int {
  var i, j, temp;
  i := 0;
  while i < 10 {
    j := 0;
    while j < 5 {
      temp := i * j;
      if temp > 20 {
        if temp < 50 {
          sum := sum + temp;
        } else {
          sum := sum - temp;
        }
      } else {
        sum := sum + (temp * 2);
      }
      j := j + 1;
    }
    i := i + 1;
  }
  return sum;
}

{
  sum := 0;
  count := 0;
  flag := 1;

  result1 := calculate(5, 3);
  result2 := process();
  result3 := result1 + result2;

  for (count := 0; count < MAX; count := count + STEP) {
    if flag = 1 {
      sum := sum + count;
      if sum > 500 {
        flag := 0;
      }
    } else {
      sum := sum - count;
    }
  }

  avg := sum / MAX;
  result1 := avg + result3;
  result2 := result1 * 2;
  result3 := result2 / 3;

  if result3 > 100 {
    if result3 < 200 {
      result1 := result3 + 50;
    } else {
      result1 := result3 - 25;
    }
  } else {
    result1 := result3 * 2;
  }
}
end
